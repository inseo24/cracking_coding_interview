```jsx
function solution(a, b, c ){
    let answer = 0;
    let cnt = 0;
    let sorted = a.sort();

    for (let i = 0; i < b; i++){      
      if (cnt == 3){
        cnt = 0;
        answer += sorted[sorted.length -2]
      } else {
        answer += sorted[sorted.length -1]
        cnt++;  
      }
      
    }

    return answer;
}
```

2. 숫자 카드 게임

```jsx
function solution(a){
    let answer = 0;
    let len = a.length;
    
    for (let i = 0; i < len; i++){
      let sorted = a[i].sort()
      let max = sorted[0]
      if ( max > answer ) answer = max;
    }

    return answer;
}

let a = [[3,1,2],[4,1,4],[2,2,2]];
console.log(solution(a));
```


3. 1이 될 때까지

```jsx
function solution(a, b){
    let answer = 0;
    
    while (a !== 1){
      if (a % b === 0){
        a = a/b
        answer++;
      } else { a -= 1; answer++; }
    }

    return answer;
}

let a = 25;
let b = 5
console.log(solution(a, b));
```
